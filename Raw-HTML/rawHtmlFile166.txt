			<div class="article-text">
									<p>Maryland will allow up to 10% capacity at the two stadiums in the state that are home to NFL teams, Gov. Larry Hogan said Friday, meaning several thousand fans could attend games.</p>

<p>The announcement affects M&amp;T Bank Stadium, where the Baltimore Ravens play, and FedEx Field in Landover, Maryland, where the Washington Football Team plays.</p>

<p>M&amp;T Bank Stadium has a total capacity of 71,008 and FedEx Field has a capacity of 82,000.</p>

<p>Neither team plays at home this weekend. The next home game for the Ravens is Nov. 1 against the Pittsburgh Steelers. The next home game for Washington is Oct. 25 against the Dallas Cowboys.</p>

<p><em><a href="https://www.gamingtoday.com/race_sports/nfl/articles"><strong>Check Out More NFL&nbsp;Content Here</strong></a></em></p>

<p>Face coverings and physical distancing will be required. Social gatherings such as tailgating will be prohibited.</p>

<p>The stadiums have been limited to 250 spectators at recent games. Those seats have been set aside mostly for players&rsquo; relatives.</p>

<p>The Republican governor wrote on Twitter he is taking the steps as Maryland&rsquo;s key health metrics for the coronavirus remain &ldquo;low and stable.&rdquo;</p>

			updateWageringBoard(dateSelected, autoRefresh, showFirstColumn);
		});

		// Auto refresh
		$('#auto-refresh').click(function(e) {
			e.preventDefault();
			onOffRefresh(dateSelected, showFirstColumn);
		});

		if (autoRefresh == 1) {
			onOffRefresh(dateSelected, showFirstColumn);
		}

		var scroller = scoreScroller(simpleBar);
		$('.wagering-data').on('mouseover', function(e) {
			clearInterval(scroller);
		});

		$('.wagering-data').on('mouseout', function(e) {
			scroller = scoreScroller(simpleBar);
		});
	});

	function scoreScroller(simpleBar) {
		var scrollEl = simpleBar.getScrollElement();
		var scroller = setInterval(function() {
			if (!window.msCrypto) {
					var pos = scrollEl.scrollTop;
					scrollEl.scrollTo(0, ++pos);
				}
			}, 150);
		if (!$('.expand-board').hasClass('toggle-shrink')) {
			clearInterval(scroller);
		}

		return scroller;
	}

	var timerRefresh = 0;

	function onOffRefresh(dateSelected, showFirstColumn) {
		if ($('.on-off-btn').html()=='Off') {
			$('.on-off-btn').html('On').css("background","#5cb85c");
			timerRefresh = setInterval(function(){
				updateWageringBoard(dateSelected, 1, showFirstColumn);
			},1000*90);
		} else {
			$('.on-off-btn').html('Off').css("background","#ffb732");;
			clearInterval(timerRefresh);
			updateWageringBoard(dateSelected, 0, showFirstColumn);
		}
	}

	function updateDefaultGameType(leagueVal) {
		var defaultGameType = 'spread_totals';
		// for MLB (5) and NHL (7) show "Money Lines" by default
		if (leagueVal == 5 || leagueVal == 7) {
			defaultGameType = 'money_lines';
		}
		$('.gametype').val(defaultGameType);
		return defaultGameType;
	}

	function updateWageringBoard(dateSelected, refresh, showFirstColumn, updateLeague, updateTimezone) {
		var leagueVal= $('.leagueselect option:selected').val();
		var typeVal;
		var expand = true;
		var timezoneVal = $('.timezone option:selected').val();

		if (leagueVal == 5) {
			$('.mlb').removeClass('hidden');
			$('.others').addClass('hidden');
			typeVal = $('.mlb option:selected').val();
		} else {
			$('.mlb').addClass('hidden');
			$('.others').removeClass('hidden');
			typeVal = $('.others option:selected').val();
		}

		if (updateLeague === true) {
			typeVal = updateDefaultGameType(leagueVal);

			// set the league selection as session default
			sessionStorage.setItem('defaultLeague', leagueVal);
		}

		// set the timezone selection as session default
		if (updateTimezone === true) {
			sessionStorage.setItem('defaultTimezone', timezoneVal);
		}

		$('.datepicker').data('DateTimePicker').date(dateSelected);

		if ($('.expand-board').length > 0) {
			var paramsObj = {
				'league' : leagueVal,
				'type' : typeVal,
				'timezone' : timezoneVal,
				'date' : dateSelected,
				'autorefresh' : refresh
			};

			var params = $.param(paramsObj);
			var href = $('.expand-board').attr('href');
			var index = href.indexOf('?')
			if (index >= 0) {
				href = href.substr(0, index)
			}

			href += '?' + params;
			$('.expand-board').attr('href', href);
		}

		$.ajax({
			url: '/odd/index',
			type: 'get',
			data: {
				'league': leagueVal,
				'type': typeVal,
				'date': dateSelected,
				'timezone': timezoneVal,
				'autorefresh': refresh,
				'expand': expand
			},
			dataType: 'html',
			success: function(html) {
				$('.wagering-data').empty().html(html);
				if (!expand) {
					$('.wagering-score').remove();
				}
				showFirstColumn();
				$('.datepicker-group span:not(.glyphicon)').empty().text(moment(dateSelected).format('MM/DD/YYYY'));
				$('.previous-link').attr('data-autorefresh', refresh);
				$('.previous-link').data('autorefresh', refresh);
				$('.previous-link').attr('data-date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.previous-link').data('date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').attr('data-autorefresh', refresh);
				$('.next-link').data('autorefresh', refresh);
				$('.next-link').attr('data-date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').data('date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
			}
		});
	}

	function initSimpleBar() {
		return new SimpleBar($('.wagering-wrap')[0]);
	}

				<h1 class="article-title">Maryland allowing up to 10% capacity at NFL stadiums</h1>
					<div class="featured-img-wrap">
				<img src="https://d2u0l2ivr9qcwr.cloudfront.net/webroot/files/articles/93587/NFL_MT_Stadium_2007_Flickr_Sam_Lehman_615x458.jpg" alt="Maryland allowing up to 10% capacity at NFL stadiums" class="img-responsive article-feature"/>									<div class="featured-img-caption">
						M&T Bank Stadium, home of the Baltimore Ravens  -Flickr/Sam Lehman					</div>
							</div>
						<div class="article-content">
			<div class="row">
				<div class="col-xs-12">
					
					<hr class="article-meta">
					<div class="row">
						<div class="col-sm-6 col-md-8">
							Oct 16, 2020 5:59 PM<br>
<strong>
	<a href="/columnist/bio/516" rel="author">Staff &amp; Wire Reports</a></strong>

											<div class="display-tags">
			<span>
			<a href="/articles/index/tag:NFL">NFL</a>		</span>

			<div class="article-text">
									<p>Maryland will allow up to 10% capacity at the two stadiums in the state that are home to NFL teams, Gov. Larry Hogan said Friday, meaning several thousand fans could attend games.</p>

<p>The announcement affects M&amp;T Bank Stadium, where the Baltimore Ravens play, and FedEx Field in Landover, Maryland, where the Washington Football Team plays.</p>

<p>M&amp;T Bank Stadium has a total capacity of 71,008 and FedEx Field has a capacity of 82,000.</p>

<p>Neither team plays at home this weekend. The next home game for the Ravens is Nov. 1 against the Pittsburgh Steelers. The next home game for Washington is Oct. 25 against the Dallas Cowboys.</p>

<p><em><a href="https://www.gamingtoday.com/race_sports/nfl/articles"><strong>Check Out More NFL&nbsp;Content Here</strong></a></em></p>

<p>Face coverings and physical distancing will be required. Social gatherings such as tailgating will be prohibited.</p>

<p>The stadiums have been limited to 250 spectators at recent games. Those seats have been set aside mostly for players&rsquo; relatives.</p>

<p>The Republican governor wrote on Twitter he is taking the steps as Maryland&rsquo;s key health metrics for the coronavirus remain &ldquo;low and stable.&rdquo;</p>

	function scoreScroller(simpleBar) {
		var scrollEl = simpleBar.getScrollElement();
		var scroller = setInterval(function() {
			if (!window.msCrypto) {

	}

	var timerRefresh = 0;

	function onOffRefresh(dateSelected, showFirstColumn) {
		if ($('.on-off-btn').html()=='Off') {
			$('.on-off-btn').html('On').css("background","#5cb85c");
			timerRefresh = setInterval(function(){
				updateWageringBoard(dateSelected, 1, showFirstColumn);
			},1000*90);
		} else {
			$('.on-off-btn').html('Off').css("background","#ffb732");;
			clearInterval(timerRefresh);
			updateWageringBoard(dateSelected, 0, showFirstColumn);
		}
	}

	function updateDefaultGameType(leagueVal) {
		var defaultGameType = 'spread_totals';
		// for MLB (5) and NHL (7) show "Money Lines" by default
		if (leagueVal == 5 || leagueVal == 7) {
			defaultGameType = 'money_lines';
		}
		$('.gametype').val(defaultGameType);
		return defaultGameType;
	}

	function updateWageringBoard(dateSelected, refresh, showFirstColumn, updateLeague, updateTimezone) {
		var leagueVal= $('.leagueselect option:selected').val();
		var typeVal;
		var expand = true;
		var timezoneVal = $('.timezone option:selected').val();

		if (leagueVal == 5) {
			$('.mlb').removeClass('hidden');
			$('.others').addClass('hidden');
			typeVal = $('.mlb option:selected').val();
		} else {
			$('.mlb').addClass('hidden');
			$('.others').removeClass('hidden');
			typeVal = $('.others option:selected').val();
		}

		if (updateLeague === true) {
			typeVal = updateDefaultGameType(leagueVal);

			// set the league selection as session default
			sessionStorage.setItem('defaultLeague', leagueVal);
		}

		// set the timezone selection as session default
		if (updateTimezone === true) {
			sessionStorage.setItem('defaultTimezone', timezoneVal);
		}

		$('.datepicker').data('DateTimePicker').date(dateSelected);

		if ($('.expand-board').length > 0) {
			var paramsObj = {
				'league' : leagueVal,
				'type' : typeVal,
				'timezone' : timezoneVal,
				'date' : dateSelected,
				'autorefresh' : refresh
			};

			var params = $.param(paramsObj);
			var href = $('.expand-board').attr('href');
			var index = href.indexOf('?')
			if (index >= 0) {
				href = href.substr(0, index)
			}

			href += '?' + params;
			$('.expand-board').attr('href', href);
		}

		$.ajax({
			url: '/odd/index',
			type: 'get',
			data: {
				'league': leagueVal,
				'type': typeVal,
				'date': dateSelected,
				'timezone': timezoneVal,
				'autorefresh': refresh,
				'expand': expand
			},
			dataType: 'html',
			success: function(html) {
				$('.wagering-data').empty().html(html);
				if (!expand) {
					$('.wagering-score').remove();
				}
				showFirstColumn();
				$('.datepicker-group span:not(.glyphicon)').empty().text(moment(dateSelected).format('MM/DD/YYYY'));
				$('.previous-link').attr('data-autorefresh', refresh);
				$('.previous-link').data('autorefresh', refresh);
				$('.previous-link').attr('data-date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.previous-link').data('date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').attr('data-autorefresh', refresh);
				$('.next-link').data('autorefresh', refresh);
				$('.next-link').attr('data-date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').data('date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
			}
		});
	}

				<h1 class="article-title">Maryland allowing up to 10% capacity at NFL stadiums</h1>
					<div class="featured-img-wrap">
				<img src="https://d2u0l2ivr9qcwr.cloudfront.net/webroot/files/articles/93587/NFL_MT_Stadium_2007_Flickr_Sam_Lehman_615x458.jpg" alt="Maryland allowing up to 10% capacity at NFL stadiums" class="img-responsive article-feature"/>									<div class="featured-img-caption">
						M&T Bank Stadium, home of the Baltimore Ravens  -Flickr/Sam Lehman					</div>
							</div>
						<div class="article-content">
			<div class="row">
				<div class="col-xs-12">
					
					<hr class="article-meta">
					<div class="row">
						<div class="col-sm-6 col-md-8">
							Oct 16, 2020 5:59 PM<br>
<strong>
	<a href="/columnist/bio/516" rel="author">Staff &amp; Wire Reports</a></strong>
						</div>
						<div class="col-sm-6 col-md-4 text-right">
							<div class="addthis_inline_share_toolbox"></div>

					<hr class="article-meta">
											<div class="display-tags">
			<span>
			<a href="/articles/index/tag:NFL">NFL</a>		</span>

			<a href="/articles/index/tag:Baltimore%20Ravens">Baltimore Ravens</a>		</span>
	</div>
						<hr class="article-meta">
									</div>
			</div>

			<div class="article-text">
									<p>Maryland will allow up to 10% capacity at the two stadiums in the state that are home to NFL teams, Gov. Larry Hogan said Friday, meaning several thousand fans could attend games.</p>

<p>The announcement affects M&amp;T Bank Stadium, where the Baltimore Ravens play, and FedEx Field in Landover, Maryland, where the Washington Football Team plays.</p>

<p>M&amp;T Bank Stadium has a total capacity of 71,008 and FedEx Field has a capacity of 82,000.</p>

<p>Neither team plays at home this weekend. The next home game for the Ravens is Nov. 1 against the Pittsburgh Steelers. The next home game for Washington is Oct. 25 against the Dallas Cowboys.</p>

<p><em><a href="https://www.gamingtoday.com/race_sports/nfl/articles"><strong>Check Out More NFL&nbsp;Content Here</strong></a></em></p>

<p>Face coverings and physical distancing will be required. Social gatherings such as tailgating will be prohibited.</p>

<p>The stadiums have been limited to 250 spectators at recent games. Those seats have been set aside mostly for players&rsquo; relatives.</p>

<p>The Republican governor wrote on Twitter he is taking the steps as Maryland&rsquo;s key health metrics for the coronavirus remain &ldquo;low and stable.&rdquo;</p>

	function scoreScroller(simpleBar) {
		var scrollEl = simpleBar.getScrollElement();
		var scroller = setInterval(function() {
			if (!window.msCrypto) {
					var pos = scrollEl.scrollTop;
					scrollEl.scrollTo(0, ++pos);

		if (!$('.expand-board').hasClass('toggle-shrink')) {
			clearInterval(scroller);
		}

		return scroller;
	}

	var timerRefresh = 0;

	function onOffRefresh(dateSelected, showFirstColumn) {
		if ($('.on-off-btn').html()=='Off') {
			$('.on-off-btn').html('On').css("background","#5cb85c");
			timerRefresh = setInterval(function(){
				updateWageringBoard(dateSelected, 1, showFirstColumn);
			},1000*90);
		} else {
			$('.on-off-btn').html('Off').css("background","#ffb732");;
			clearInterval(timerRefresh);
			updateWageringBoard(dateSelected, 0, showFirstColumn);
		}
	}

	function updateDefaultGameType(leagueVal) {
		var defaultGameType = 'spread_totals';
		// for MLB (5) and NHL (7) show "Money Lines" by default
		if (leagueVal == 5 || leagueVal == 7) {
			defaultGameType = 'money_lines';
		}
		$('.gametype').val(defaultGameType);
		return defaultGameType;
	}

	function updateWageringBoard(dateSelected, refresh, showFirstColumn, updateLeague, updateTimezone) {
		var leagueVal= $('.leagueselect option:selected').val();
		var typeVal;
		var expand = true;
		var timezoneVal = $('.timezone option:selected').val();

		if (leagueVal == 5) {
			$('.mlb').removeClass('hidden');
			$('.others').addClass('hidden');
			typeVal = $('.mlb option:selected').val();
		} else {
			$('.mlb').addClass('hidden');
			$('.others').removeClass('hidden');
			typeVal = $('.others option:selected').val();
		}

		if (updateLeague === true) {
			typeVal = updateDefaultGameType(leagueVal);

			// set the league selection as session default
			sessionStorage.setItem('defaultLeague', leagueVal);
		}

		// set the timezone selection as session default
		if (updateTimezone === true) {
			sessionStorage.setItem('defaultTimezone', timezoneVal);
		}

		$('.datepicker').data('DateTimePicker').date(dateSelected);

		if ($('.expand-board').length > 0) {
			var paramsObj = {
				'league' : leagueVal,
				'type' : typeVal,
				'timezone' : timezoneVal,
				'date' : dateSelected,
				'autorefresh' : refresh
			};

			var params = $.param(paramsObj);
			var href = $('.expand-board').attr('href');
			var index = href.indexOf('?')
			if (index >= 0) {
				href = href.substr(0, index)
			}

			href += '?' + params;
			$('.expand-board').attr('href', href);
		}

		$.ajax({
			url: '/odd/index',
			type: 'get',
			data: {
				'league': leagueVal,
				'type': typeVal,
				'date': dateSelected,
				'timezone': timezoneVal,
				'autorefresh': refresh,
				'expand': expand
			},
			dataType: 'html',
			success: function(html) {
				$('.wagering-data').empty().html(html);
				if (!expand) {
					$('.wagering-score').remove();
				}
				showFirstColumn();
				$('.datepicker-group span:not(.glyphicon)').empty().text(moment(dateSelected).format('MM/DD/YYYY'));
				$('.previous-link').attr('data-autorefresh', refresh);
				$('.previous-link').data('autorefresh', refresh);
				$('.previous-link').attr('data-date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.previous-link').data('date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').attr('data-autorefresh', refresh);
				$('.next-link').data('autorefresh', refresh);
				$('.next-link').attr('data-date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').data('date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
			}
		});
	}

									<p>Maryland will allow up to 10% capacity at the two stadiums in the state that are home to NFL teams, Gov. Larry Hogan said Friday, meaning several thousand fans could attend games.</p>

<p>The announcement affects M&amp;T Bank Stadium, where the Baltimore Ravens play, and FedEx Field in Landover, Maryland, where the Washington Football Team plays.</p>

<p>M&amp;T Bank Stadium has a total capacity of 71,008 and FedEx Field has a capacity of 82,000.</p>

<p>Neither team plays at home this weekend. The next home game for the Ravens is Nov. 1 against the Pittsburgh Steelers. The next home game for Washington is Oct. 25 against the Dallas Cowboys.</p>

<p><em><a href="https://www.gamingtoday.com/race_sports/nfl/articles"><strong>Check Out More NFL&nbsp;Content Here</strong></a></em></p>

<p>Face coverings and physical distancing will be required. Social gatherings such as tailgating will be prohibited.</p>

<p>The stadiums have been limited to 250 spectators at recent games. Those seats have been set aside mostly for players&rsquo; relatives.</p>

<p>The Republican governor wrote on Twitter he is taking the steps as Maryland&rsquo;s key health metrics for the coronavirus remain &ldquo;low and stable.&rdquo;</p>

			sessionStorage.setItem('defaultTimezone', timezoneVal);
		}

		$('.datepicker').data('DateTimePicker').date(dateSelected);

		if ($('.expand-board').length > 0) {
			var paramsObj = {
				'league' : leagueVal,
				'type' : typeVal,
				'timezone' : timezoneVal,
				'date' : dateSelected,
				'autorefresh' : refresh
			};

			var params = $.param(paramsObj);
			var href = $('.expand-board').attr('href');
			var index = href.indexOf('?')
			if (index >= 0) {
				href = href.substr(0, index)
			}

			href += '?' + params;
			$('.expand-board').attr('href', href);
		}

		$.ajax({
			url: '/odd/index',
			type: 'get',
			data: {
				'league': leagueVal,
				'type': typeVal,
				'date': dateSelected,
				'timezone': timezoneVal,
				'autorefresh': refresh,
				'expand': expand
			},
			dataType: 'html',
			success: function(html) {
				$('.wagering-data').empty().html(html);
				if (!expand) {
					$('.wagering-score').remove();
				}
				showFirstColumn();
				$('.datepicker-group span:not(.glyphicon)').empty().text(moment(dateSelected).format('MM/DD/YYYY'));
				$('.previous-link').attr('data-autorefresh', refresh);
				$('.previous-link').data('autorefresh', refresh);
				$('.previous-link').attr('data-date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.previous-link').data('date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').attr('data-autorefresh', refresh);
				$('.next-link').data('autorefresh', refresh);
				$('.next-link').attr('data-date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
				$('.next-link').data('date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
			}
		});
	}
