    // Facebook's video API doesn't work properly on mobiles
    initMobileVideoTimer = function() {
      var isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
      var mobileTimer;

      if (!expanded || !isMobile || mobileTimeout) {
        return
      }

      if (player && player.getDuration && player.getDuration() > 1) {
        mobileTimer = player.getDuration() * 1000;
      } else {
        mobileTimer = 30000; // workaround for https://developers.facebook.com/support/bugs/338091156930372/
      }

      mobileTimeout = setTimeout(function () {
        scope.$apply(function () {
          scope.videoFinishedPlaying(scope.entry_method);
        })
      }, mobileTimer)
    }

    if(scope.entry_method.config3 == "video") {
      lazyClass = "fb-video";
    }

    if( !$this.hasClass(lazyClass) ) {
      $this.addClass(lazyClass);

            player.play();
          }

          initMobileVideoTimer();
        }

        expanded = true;
        try {
          player.play();
        } catch(e) {
        }

        initMobileVideoTimer();
      };

      scope.entry_method.onUnexpand = function () {
        expanded = false;
        try {
          player.pause();
        } catch(e) {
        }
      };
    }
  };

  if (window.FB) { // FB script already loaded
    window.fbAsyncInit();
  }

      height = Math.floor((width * 9) / 16) - heightOffset,
      player;

  $t.parent().height(height);

      }
