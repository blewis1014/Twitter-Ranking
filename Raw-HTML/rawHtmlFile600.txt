  function relativeTimestamp() {
    $('.relative-time').each(function() {
      var $time = $(this);
      var timestamp = parseInt($time.data('timestamp'));
      var content_type = $time.data('content-type');

      var now = new Date();
      var secondsPast = (now.getTime()/1000 - timestamp);
      var relative_text;

      if(secondsPast >= 0) {
        if(secondsPast <= 60) {
          relative_text = "1m ago";
        } else if(secondsPast < 3600) {
          relative_text = "%sm ago".replace('%s', parseInt(secondsPast/60));
        } else if((content_type == 'article' && secondsPast < 28800) || (content_type == 'comment' && secondsPast < 86400)) {
          relative_text = "%sh ago".replace('%s', parseInt(secondsPast/3600));
        }
      }

      if(relative_text) {
        $time.text(relative_text);
      }
    });
  }

$('table.sortable th').click(function() {
  var table = $(this).parents('table').eq(0);
  var rows = table.find('tr:gt(0)').toArray().sort(tableSortComparer($(this).index()));
  this.asc = !this.asc
  if (!this.asc){rows = rows.reverse();}
  for (var i = 0; i < rows.length; i++){table.append(rows[i]);}
});
function tableSortComparer(index) {
  return function(a, b) {
    var valA = getTableCellValue(a, index), valB = getTableCellValue(b, index);
    return $.isNumeric(valA) && $.isNumeric(valB) ? valA - valB : valA.toString().localeCompare(valB);
  }
}
function getTableCellValue(row, index){ return $(row).children('td').eq(index).data('sortBy') || $(row).children('td').eq(index).text(); }
