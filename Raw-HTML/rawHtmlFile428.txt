function buildWpEmbed(obj){
    var wp_embed = '[podbean';
    for(var i in obj){
        wp_embed+= ' '+i+'="'+obj[i]+'"';
    }
    return wp_embed+']';
}

$('.ul-skins a[data-skin]').click(function(event) {
  var skinIndex = $(this).attr('data-skin');
  var form = $(this).parents('form');
  
  var is_btn = $(this).parents('.ul-button').length>0;
  if(is_btn){
    form.find("input[name='btn-skin']").val(skinIndex);
    form.find('.ul-button').find('a.current').removeClass('current');
  }else{
    form.find("input[name='skin']").val(skinIndex);
    form.find('.ul-skins').not('.ul-button').find('a.current').removeClass('current');
  }
  
  $(this).addClass('current');

  updateIframe($(this));
});

function updateWpEmbed(form){
    var wp_code = form.attr('data-wp-code');
    var wp_domain = form.attr('data-wp-domain');
    var current_tab_id = form.parents('.tab-pane').attr('id');

    var wpdata = {};
    
    wpdata.resource = 'episode='+wp_code+'';
    if(current_tab_id=='tab4'){
        wpdata.type = 'audio-rectangle';
        wpdata.height = 100;
    }else if(current_tab_id=='tab3'){
        wpdata.type = 'audio-square';
        wpdata.height = 315;
    }else{
        wpdata.type = 'video';
        wpdata.height = form.find("input[name='size']:checked").val();
    }

    if(wp_domain){
        wpdata.domain = wp_domain;
    }
    
    form.serializeArray().forEach(function(v){
        wpdata[v.name] = encodeURIComponent(v.value);
    });

    form.parents('.audioplayer-area').find('.wp-code').val(buildWpEmbed(wpdata));
}

function updateIframe(self){
    var player = self.parents('.audioplayer-area');
    var form = self.parents('form');
    var iframe = player.find('iframe');

    var link = iframe.attr('data-link');
    var params = form.serialize();

    iframe.attr('src',link+'&'+params);
    
    if(form.find("input[name='size']").length>0){
        iframe.attr('height',form.find("input[name='size']:checked").val());
    }
    
    var box = player.find('.iframe-box').clone();

    box.find('iframe').removeAttr('data-link');

    player.find('.embedcode').val($.trim(box.html().replace(/&amp;/g,'&')));

    updateWpEmbed(form);
}
