function initYoutubeVideo() {
  var $t = $(this);

  if(!window.YT || !window.YT.Player) {
    $t.addClass('youtube-auto-init');
    return;
  }

  if (!window.devicePixelRatio && window.screen.deviceXDPI && window.screen.logicalXDPI) {
    window.devicePixelRatio = window.devicePixelRatio ||
      window.screen.deviceXDPI / window.screen.logicalXDPI;
  }

  var heightOffset = 0;
  if (window.devicePixelRatio == 1) {
    // Force hide 1px black bar on standard-res devices
    heightOffset = 3;
  }

  var width = $('.popup-blocks-container').width(),
      height = Math.floor((width * 9) / 16) - heightOffset,
      player;

  $t.parent().height(height);

  var playerVars = { controls: 1, disablekb: 0, rel: 0, modestbranding: 1, showinfo: 1, autoplay: 0 };

  var scope = angular.element(this).scope();
  if(!scope) return;

  var videoId, callback, postInit;

  if(scope.entry_method) {
    scope.entry_method.onExpand = function () {
      try {
        if (!navigator.userAgent.match(/iPod|iPhone/) && ((scope.entry_method.config4 + '').match(/autoplay=1/i) || player.getCurrentTime() > 0)) {
          player.playVideo();
        }
      } catch (e) {
      }
    };

    scope.entry_method.onUnexpand = function () {
      try {
        player.pauseVideo();
      } catch (e) {
      }
    };

    var pairs = (scope.entry_method.config4+'').split(/&/);
    var pair;

    for(var i = 0; i < pairs.length; i++) {
      pair = pairs[i].split('=');
      if(pair.length == 2 && pair[0].toLowerCase() != 'autoplay') {
        playerVars[pair[0]] = pair[1];
      }
    }
    videoId = scope.entry_method.config3;
    callback  = function (event) {
      scope.youtubeVideoStateChange(scope.entry_method, event);
      scope.$apply();
    };

    postInit = function () {
      var grace = 5;
      var secondsOpen;
      var interval = setInterval(function () {
        secondsOpen = scope.entry_method.secondsOpen || 0;

        if(player && player.getDuration && player.getDuration() >  1 && (player.getDuration() > grace || secondsOpen > player.getDuration())) {
          if(player.getCurrentTime() > (player.getDuration() - grace) || secondsOpen > player.getDuration()) {
            scope.$apply(function () {
              clearInterval(interval);
              setTimeout(function () {
                scope.entry_method.timePassed = true
              }, (grace-1) * 1000);
            });
          }
        }
      }, 1000);
    }
  } else {
    videoId = scope.incentive.youtube_video_id;
    callback = postInit = function() {}
  }

  if(window.YT && window.YT.Player) {
    player = new YT.Player(this.id, {
      height: height,
      width: width,
      videoId: videoId,
      playerVars: playerVars,
      events: {onStateChange: callback}
    });

    postInit();
  }
}

        player.pauseVideo();
